
0. Start mysql

 Local
 > mysql -u root -p
 Openshift
 > mysql

1. Use Database

 > USE clabkiapi;

2. Create Table

 > CREATE TABLE status(id INT NOT NULL PRIMARY KEY AUTO_INCREMENT,major INT(16), minor INT(16), reported_as_lost BOOL, user CHAR(20));


3. Create index

 > CREATE UNIQUE INDEX major_minor_combination ON status (major,minor);


4. Check Table

 > DESCRIBE status;

5. INSERT

  > INSERT INTO 'status' ('id','major','minor','reported_as_lost','user') VALUES (0,1,1,false,"nico");
